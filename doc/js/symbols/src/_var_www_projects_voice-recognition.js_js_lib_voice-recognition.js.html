1 /**
  2  * A voice model.
  3  * @constructor
  4  * @param {Object} data The model data containing the name, the speaker's name & gender, the microphone's name and the voice data.
  5  */
  6 var VoiceModel = function VoiceModel(data) {
  7 	this._name = data.name; //Model's name
  8 	this._gender = data.gender; //Speaker's gender
  9 	this._speaker = data.speaker; //Speaker's name
 10 	this._micro = data.micro; //Speaker's microphone
 11 
 12 	//Voice data
 13 	this._data = {
 14 		magnitude: data.data.magnitude,
 15 		time: data.data.time,
 16 		frequencies: data.data.frequencies
 17 	};
 18 };
 19 VoiceModel.prototype = {
 20 	/**
 21 	 * Get the model's name.
 22 	 * @return {String}
 23 	 */
 24 	name: function getName() {
 25 		return this._name;
 26 	},
 27 	/**
 28 	 * Get the speaker's gender.
 29 	 * @return {String}
 30 	 */
 31 	gender: function getGender() {
 32 		return this._gender;
 33 	},
 34 	/**
 35 	 * Get the speaker's name.
 36 	 * @return {String}
 37 	 */
 38 	speaker: function getSpeaker() {
 39 		return this._speaker;
 40 	},
 41 	/**
 42 	 * Get the microphone's name.
 43 	 * @return {String}
 44 	 */
 45 	micro: function getMicro() {
 46 		return this._micro;
 47 	},
 48 	/**
 49 	 * Get this model's range (when does the speaker speak).
 50 	 * @return {Number[]} An array containing two indexes : the begining & the end.
 51 	 */
 52 	range: function getRange() {
 53 		return [0, this.standardizedData().magnitude.length - 1];
 54 	},
 55 	/**
 56 	 * Get this model's standardized data.
 57 	 * @return {Object}
 58 	 */
 59 	standardizedData: function getStandardizedData() {
 60 		return {
 61 			magnitude: this._data.magnitude,
 62 			time: this._data.time
 63 		};
 64 	},
 65 	/**
 66 	 * Get frequencies on which this model is made.
 67 	 * @return {Number[]} Frequencies on which this model is made.
 68 	 */
 69 	frequencies: function frequencies() {
 70 		return this._data.frequencies;
 71 	},
 72 	/**
 73 	 * Get this model's status.
 74 	 * @return {Number}
 75 	 */
 76 	status: function getStatus() {
 77 		return 4;
 78 	}
 79 };

 82 /**
 83  * A voice recognition.
 84  * @constructor
 85  */
 86 var VoiceRecognition = function VoiceRecognition() {
 87 	Utils.Observable.call(this); // Inheritance from Observable
 88 
 89 	//Initialize properties
 90 	this._status = 0;
 91 	this._input = null;
 92 	this._models = [];
 93 
 94 	this._data = null;
 95 	this._result = null;
 96 };
 97 VoiceRecognition.prototype = {
 98 	/**
 99 	 * Get the recognition's status.
100 	 */
101 	status: function getStatus() {
102 		return this._status;
103 	},
104 	/**
105 	 * Update this comparison's status.
106 	 * @param  {Number} [status] The new status.
107 	 * @private
108 	 */
109 	_updateStatus: function _updateStatus(status) {
110 		status = (typeof status == 'number') ? status : this._status;
111 		this._status = status;
112 
113 		//Trigger the event
114 		this.notify('updatestatus', { status: status });
115 	},
116 	/**
117 	 * Set the analysis which will be recognized.
118 	 * @param {VoiceAnalysis} analysis The voice analysis.
119 	 */
120 	setInputAnalysis: function setInputAnalysis(analysis) {
121 		this._input = analysis;
122 
123 		//Trigger the event
124 		this.notify('inputchange', {
125 			analysis: analysis
126 		});
127 	},
128 	/**
129 	 * Set the voice models, with which the voice analysis will be compared.
130 	 * @param {VoiceModel[]} models An array of models.
131 	 */
132 	setVoiceModels: function setVoiceModels(models) {
133 		this._models = models;
134 
135 		//Trigger the events
136 		this.notify('modelschange', {
137 			models: models
138 		});
139 	},
140 	/**
141 	 * Get the number of voice models selected.
142 	 * @return {Number} The number of voice models.
143 	 */
144 	countVoiceModels: function countVoiceModels() {
145 		var nbr = 0;
146 
147 		for (var i = 0; i < this._models.length; i++) { //For each model set
148 			nbr += this._models[i].models.length;
149 		}
150 
151 		return nbr;
152 	},
153 	/**
154 	 * Start the voice recognition.
155 	 * @return {Object} An object containing the recognition's results.
156 	 */
157 	recognize: function recognize() {
158 		this.notify('start');
159 
160 		Utils.logMessage('Recognition start');
161 		Utils.logMessage('---------');
162 
163 		var that = this, analysis = this._input;
164 
165 		var avgs = [], stds = [];
166 		var modelSetsAvgs = [], modelSetsStds = [], modelSetMinAvg, modelSetIndex;
167 
168 		for (var i = 0; i < this._models.length; i++) { //For each model set
169 			(function(modelSet) {
170 				avgs[i] = [];
171 				stds[i] = [];
172 
173 				for (var j = 0; j < modelSet.models.length; j++) { //For each voice model in the model set
174 					(function(modelData) {
175 						var model = new VoiceModel(modelData); //Build a new voice model
176 
177 						var comparison = VoiceComparison.build(model, analysis); //... And a new voice comparison
178 
179 						that.notify('comparestart', {
180 							model: model,
181 							comparison: comparison
182 						});
183 
184 						var result = comparison.compareData(); //Compare the model & the analysis
185 
186 						if (result === false) { //An error occured
187 							that.notify('compareerror', {
188 								model: model,
189 								comparison: comparison
190 							});
191 							return;
192 						}
193 
194 						avgs[i][j] = result.avg;
195 						stds[i][j] = result.std;
196 
197 						that.notify('comparecomplete', {
198 							model: model,
199 							comparison: comparison
200 						});
201 
202 					})(modelSet.models[j]);
203 				}
204 
205 				//Store calculated data
206 				modelSetsAvgs[i] = Utils.Math.getAverageFromNumArr(avgs[i]);
207 				modelSetsStds[i] = Utils.Math.getAverageFromNumArr(stds[i]);
208 
209 				if (typeof modelSetMinAvg == 'undefined' || modelSetsAvgs[i] < modelSetMinAvg) { //If it's the lowest deviation average
210 					modelSetMinAvg = modelSetsAvgs[i];
211 					modelSetIndex = i;
212 				}
213 			})(this._models[i]);
214 		}
215 
216 		//Finally, create an object containing all data
217 		this._data = {
218 			models: {
219 				avgs: avgs,
220 				stds: stds
221 			},
222 			modelSets: {
223 				avgs: modelSetsAvgs,
224 				stds: modelSetsStds
225 			}
226 		};
227 
228 		//Duplicate this._data.modelSets.avgs in a new array
229 		var modelsAvgs = [];
230 		for (var i = 0; i < this._data.modelSets.avgs.length; i++) {
231 			modelsAvgs.push(this._data.modelSets.avgs[i]);
232 		}
233 
234 		modelsAvgs.sort(); //Sort values in this array
235 
236 		//Determine the error average
237 		this._stats = {
238 			avgError: Utils.Math.getNumWithSetDec(modelsAvgs[0] / modelsAvgs[1])
239 		};
240 
241 		//Final result
242 		var modelSetName = this._models[modelSetIndex].name;
243 		this._result = modelSetIndex;
244 
245 		Utils.logMessage('Recognition complete');
246 		Utils.logMessage('---------');
247 		Utils.logMessage(this._data);
248 
249 		//Trigger the event
250 		this.notify('complete', {
251 			data: this._data,
252 			stats: this._stats,
253 			result: {
254 				index: modelSetIndex,
255 				name: modelSetName
256 			}
257 		});
258 
259 		return {
260 			data: this._data,
261 			stats: this._stats,
262 			result: {
263 				index: modelSetIndex,
264 				name: modelSetName
265 			}
266 		}
267 	}
268 };
